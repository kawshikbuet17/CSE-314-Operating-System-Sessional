	#1 is the current directory

directory_tree_printing(){
	cd "$1"
	for f in *
	do
		if [ -d "$f" ]; then
			x=$(expr $2 + 1)
			print_in_level $2 "$f"
			directory_tree_printing "$f" $x

			print_in_level $2 "$f"
		fi
	done
	cd ../
}

	echo $1
	directory_tree_printing $1 1

	echo ..
	directory_tree_printing . 1
